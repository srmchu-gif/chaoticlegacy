    async function populateDecks(selectEl) {
      try {
        const res = await fetch('/api/decks');
        const data = await res.json();
        selectEl.innerHTML = '';
        if (data.decks && data.decks.length) {
          data.decks.forEach(deck => {
            const opt = document.createElement('option');
            opt.value = deck.name;
            opt.textContent = deck.name;
            selectEl.appendChild(opt);
          });
        } else {
          const opt = document.createElement('option');
          opt.value = "";
          opt.textContent = "Nenhum deck encontrado";
          selectEl.appendChild(opt);
        }
      } catch (err) {
        console.error(err);
      }
    }

      const deckData = await deckRes.json();
  
      try {
        const res = await fetch(\`/api/multiplayer/rooms/\${selectedRoomIdToJoin}/join\`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ spectator: false, deck: deckData })
        });
        
        if (!res.ok) {

          alert('Erro: ' + errorData.error);
          return;
        }
